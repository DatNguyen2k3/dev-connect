WITH 
    Point2D_with_id as (
        SELECT *, ROW_NUMBER() OVER() id
        FROM Point2D
    )

SELECT 
    ROUND(
        SQRT(
            MIN(
                POWER((p1.x-p2.x),2) + POWER((p1.y-p2.y),2)
            )
        ), 
    2) AS 'shortest'
FROM 
    Point2D_with_id p1
    JOIN Point2D_with_id p2 ON p1.id > p2.id
